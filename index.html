function getAvailableSlotsByDay(dateString) { const day = new Date(dateString).getDay(); // 日曜 if (day === 0) return []; // 平日（月〜金） if (day >= 1 && day <= 5) { const slots = []; for (let h = 7; h <= 11; h++) { slots.push(`${String(h).padStart(2, '0')}:00`); } for (let h = 14; h <= 22; h++) { slots.push(`${String(h).padStart(2, '0')}:00`); } return slots; } // 土曜 if (day === 6) { const slots = []; for (let h = 8; h <= 12; h++) { slots.push(`${String(h).padStart(2, '0')}:00`); } for (let h = 13; h <= 19; h++) { slots.push(`${String(h).padStart(2, '0')}:00`); } return slots; } return []; }
